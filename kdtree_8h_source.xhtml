19 // Part of the implementation of the Super 4-points Congruent Sets (Super 4PCS)
20 // algorithm presented in:
21 //
22 // Super 4PCS: Fast Global Pointcloud Registration via Smart Indexing
23 // Nicolas Mellado, Dror Aiger, Niloy J. Mitra
24 // Symposium on Geometry Processing 2014.
25 //
26 // Data acquisition in large-scale scenes regularly involves accumulating
27 // information across multiple scans. A common approach is to locally align scan
28 // pairs using Iterative Closest Point (ICP) algorithm (or its variants), but
29 // requires static scenes and small motion between scan pairs. This prevents
30 // accumulating data across multiple scan sessions and/or different acquisition
31 // modalities (e.g., stereo, depth scans). Alternatively, one can use a global
32 // registration algorithm allowing scans to be in arbitrary initial poses. The
33 // state-of-the-art global registration algorithm, 4PCS, however has a quadratic
34 // time complexity in the number of data points. This vastly limits its
35 // applicability to acquisition of large environments. We present Super 4PCS for
36 // global pointcloud registration that is optimal, i.e., runs in linear time (in
37 // the number of data points) and is also output sensitive in the complexity of
38 // the alignment problem based on the (unknown) overlap across scan pairs.
39 // Technically, we map the algorithm as an ‘instance problem’ and solve it
40 // efficiently using a smart indexing data organization. The algorithm is
41 // simple, memory-efficient, and fast. We demonstrate that Super 4PCS results in
42 // significant speedup over alternative approaches and allows unstructured
43 // efficient acquisition of scenes at scales previously not possible. Complete
44 // source code and datasets are available for research use at
45 // http://geometry.cs.ucl.ac.uk/projects/2014/super4PCS/.

394 template<typename Scalar, typename Index>
395 Index
396 KdTree<Scalar, Index>::doQueryRestrictedClosestIndex(
397  const VectorType& queryPoint,
398  Scalar sqdist,
399  int currentId)
400 {
401 
402  Index cl_id = invalidIndex();
403  Scalar cl_dist = sqdist;
404 
405  mNodeStack[0].nodeId = 0;
406  mNodeStack[0].sq = 0.f;
407  unsigned int count = 1;
408 
409  //int nbLoop = 0;
410  while (count)
411  {
412  //nbLoop++;
413  QueryNode& qnode = mNodeStack[count-1];
414  KdNode & node = mNodes[qnode.nodeId];
415 
416  if (qnode.sq < cl_dist)
417  {
418  if (node.leaf)
419  {
420  --count; // pop
421  const int end = node.start+node.size;
422  for (int i=node.start ; i<end ; ++i){
423  const Scalar sqdist = (queryPoint - mPoints[i]).squaredNorm();
424  if (sqdist <= cl_dist && mIndices[i] != currentId){
425  cl_dist = sqdist;
426  cl_id = mIndices[i];
427  }
428  }
429  }
430  else
431  {
432  // replace the stack top by the farthest and push the closest
433  const Scalar new_off = queryPoint[node.dim] - node.splitValue;
434 
435  //std::cout << "new_off = " << new_off << std::endl;
436 
437  if (new_off < 0.)
438  {
439  mNodeStack[count].nodeId = node.firstChildId; // stack top the farthest
440  qnode.nodeId = node.firstChildId+1; // push the closest
441  }
442  else
443  {
444  mNodeStack[count].nodeId = node.firstChildId+1;
445  qnode.nodeId = node.firstChildId;
446  }
447  mNodeStack[count].sq = qnode.sq;
448  qnode.sq = new_off*new_off;
449  ++count;
450  }
451  }
452  else
453  {
454  // pop
455  --count;
456  }
457  }
458  return cl_id;
459 }

484  if (qnode.sq < sqdist)

488  --count; // pop

492  f(i);

495  else

522 template<typename Scalar, typename Index>
523 unsigned int KdTree<Scalar, Index>::split(int start, int end, unsigned int dim, Scalar splitValue)
524 {
525  int l(start), r(end-1);
526  for ( ; l<r ; ++l, --r)
527  {
528  while (l < end && mPoints[l][dim] < splitValue)
529  l++;
530  while (r >= start && mPoints[r][dim] >= splitValue)
531  r--;
532  if (l > r)
533  break;
534  std::swap(mPoints[l],mPoints[r]);
535  std::swap(mIndices[l],mIndices[r]);
536  }
537  return (mPoints[l][dim] < splitValue ? l+1 : l);
538 }

560 template<typename Scalar, typename Index>
561 void KdTree<Scalar, Index>::createTree(unsigned int nodeId, unsigned int start, unsigned int end, unsigned int level, unsigned int targetCellSize, unsigned int targetMaxDepth)
562 {
563 
564  KdNode& node = mNodes[nodeId];
565  AxisAlignedBoxType aabb;
566  //aabb.Set(mPoints[start]);
567  for (unsigned int i=start ; i<end ; ++i)
568  aabb.extendTo(mPoints[i]);
569 
570  VectorType diag = Scalar(0.5) * (aabb.max()- aabb.min());
571  typename VectorType::Index dim;
572 
573 #ifdef DEBUG
574 
575 // std::cout << "createTree("
576 // << nodeId << ", "
577 // << start << ", "
578 // << end << ", "
579 // << level << ")"
580 // << std::endl;
581 
582  if (std::isnan(diag.maxCoeff(&dim))){
583  std::cerr << "NaN values discovered in the tree, abort" << std::endl;
584  return;
585  }
586 #else
587  diag.maxCoeff(&dim);
588 #endif
589 
590 
591 #undef DEBUG
592  node.dim = dim;
593  node.splitValue = aabb.center()(dim);
594 
595  unsigned int midId = split(start, end, dim, node.splitValue);
596 
597  node.firstChildId = mNodes.size();
598 
599  {
600  KdNode n;
601  n.size = 0;
602  mNodes.push_back(n);
603  mNodes.push_back(n);
604  }
605  //mNodes << Node() << Node();
606  //mNodes.resize(mNodes.size()+2);
607 
608  {
609  // left child
610  unsigned int childId = mNodes[nodeId].firstChildId;
611  KdNode& child = mNodes[childId];
612  if (midId-start <= targetCellSize || level>=targetMaxDepth)
613  {
614  child.leaf = 1;
615  child.start = start;
616  child.size = midId-start;
617  }
618  else
619  {
620  child.leaf = 0;
621  createTree(childId, start, midId, level+1, targetCellSize, targetMaxDepth);
622  }
623  }
624 
625  {
626  // right child
627  unsigned int childId = mNodes[nodeId].firstChildId+1;
628  KdNode& child = mNodes[childId];
629  if (end-midId <= targetCellSize || level>=targetMaxDepth)
630  {
631  child.leaf = 1;
632  child.start = midId;
633  child.size = end-midId;
634  }
635  else
636  {
637  child.leaf = 0;
638  createTree(childId, midId, end, level+1, targetCellSize, targetMaxDepth);
639  }
640  }
641 }
