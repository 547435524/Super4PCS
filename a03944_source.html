19 // This file is part of the implementation of the 4-points Congruent Sets (4PCS)
20 // algorithm presented in:
21 //
22 // Super 4PCS: Fast Global Pointcloud Registration via Smart Indexing
23 // Nicolas Mellado, Dror Aiger, Niloy J. Mitra
24 // Symposium on Geometry Processing 2014.
25 //
26 // Given two sets of points in 3-space, P and Q, the algorithm applies RANSAC
27 // in roughly O(n^2) time instead of O(n^3) for standard RANSAC, using an
28 // efficient method based on invariants, to find the set of all 4-points in Q
29 // that can be matched by rigid transformation to a given set of 4-points in P
30 // called a base. This avoids the need to examine all sets of 3-points in Q
31 // against any base of 3-points in P as in standard RANSAC.
32 // The algorithm can use colors and normals to speed-up the matching
33 // and to improve the quality. It can be easily extended to affine/similarity
34 // transformation but then the speed-up is smaller because of the large number
35 // of congruent sets. The algorithm can also limit the range of transformations
36 // when the application knows something on the initial pose but this is not
37 // necessary in general (though can speed the runtime significantly).
38 
39 // Home page of the 4PCS project (containing the paper, presentations and a
40 // demo): http://graphics.stanford.edu/~niloy/research/fpcs/fpcs_sig_08.html
41 // Use google search on "4-points congruent sets" to see many related papers
42 // and applications.

119 static inline int getArgs(int argc,
120  char **argv,
121  const Utils::Logger& logger = Utils::Logger()) {
122  int i = 1;
123  while (i < argc) {
124  if (!strcmp(argv[i], "-i")) {
125  input1 = std::string(argv[++i]);
126  input2 = std::string(argv[++i]);
127  } else if (!strcmp(argv[i], "-o")) {
128  overlap = atof(argv[++i]);
129  } else if (!strcmp(argv[i], "-d")) {
130  delta = atof(argv[++i]);
131  } else if (!strcmp(argv[i], "-c")) {
132  max_color = atof(argv[++i]);
133  } else if (!strcmp(argv[i], "-t")) {
134  max_time_seconds = atoi(argv[++i]);
135  } else if (!strcmp(argv[i], "-a")) {
136  norm_diff = atof(argv[++i]);
137  } else if (!strcmp(argv[i], "-n")) {
138  n_points = atoi(argv[++i]);
139  } else if (!strcmp(argv[i], "-r")) {
140  output = argv[++i];
141  } else if (!strcmp(argv[i], "-m")) {
142  outputMat = argv[++i];
143  } else if (!strcmp(argv[i], "-x")) {
144  use_super4pcs = false;
145  } else if (!strcmp(argv[i], "--sampled1")) {
146  outputSampled1 = argv[++i];
147  } else if (!strcmp(argv[i], "--sampled2")) {
148  outputSampled2 = argv[++i];
149  } else if (!strcmp(argv[i], "-h")) {
150  return 1;
151  } else if (argv[i][0] == '-') {
152  std::cerr << "Unknown flag\n";
153  return -1;
154  };
155  i++;
156  }
157 
158  // if no output file (geometry/matrix) is set, force 3d mesh
159  if (output.empty() && outputMat.empty()) output = defaultObjOutput;
160 
161  return 0;
162 }

165  const Utils::Logger& logger = Utils::Logger())
166 {
167  bool overlapOk = options.configureOverlap(overlap);
168  if(! overlapOk ) {
169  logger.Log<Utils::ErrorReport>("Invalid overlap configuration. ABORT");
170  return false;
171  }
172  options.sample_size = n_points;
173  options.max_normal_difference = norm_diff;
174  options.max_color_distance = max_color;
175  options.max_time_seconds = max_time_seconds;
176  options.delta = delta;
177 
178  return true;
179 }
