19 // An implementation of the 4-points Congruent Sets (4PCS) algorithm presented
20 // in:
21 //
22 // 4-points Congruent Sets for Robust Surface Registration
23 // Dror Aiger, Niloy J. Mitra, Daniel Cohen-Or
24 // ACM SIGGRAPH 2008 and ACM Transaction of Graphics.
25 //
26 // Given two sets of points in 3-space, P and Q, the algorithm applies RANSAC
27 // in roughly O(n^2) time instead of O(n^3) for standard RANSAC, using an
28 // efficient method based on invariants, to find the set of all 4-points in Q
29 // that can be matched by rigid transformation to a given set of 4-points in P
30 // called a base. This avoids the need to examine all sets of 3-points in Q
31 // against any base of 3-points in P as in standard RANSAC.
32 // The algorithm can use colors and normals to speed-up the matching
33 // and to improve the quality. It can be easily extended to affine/similarity
34 // transformation but then the speed-up is smaller because of the large number
35 // of congruent sets. The algorithm can also limit the range of transformations
36 // when the application knows something on the initial pose but this is not
37 // necessary in general (though can speed the runtime significantly).
38 
39 // Home page of the 4PCS project (containing the paper, presentations and a
40 // demo): http://graphics.stanford.edu/~niloy/research/fpcs/fpcs_sig_08.html
41 // Use google search on "4-points congruent sets" to see many related papers
42 // and applications.

116 // Holds a base from P. The base contains 4 points (indices) from the set P.
117 struct Quadrilateral {
118  std::array <int, 4> vertices;
119  inline Quadrilateral(int vertex0, int vertex1, int vertex2, int vertex3) {
120  vertices = { vertex0, vertex1, vertex2, vertex3 };
121  }
122 
123  inline bool operator< (const Quadrilateral& rhs) const {
124  return vertices[0] != rhs[0] ? vertices[0] < rhs[0]
125  : vertices[1] != rhs[1] ? vertices[1] < rhs[1]
126  : vertices[2] != rhs[2] ? vertices[2] < rhs[2]
127  : vertices[3] < rhs[3];
128  }
129 
130  inline bool operator== (const Quadrilateral& rhs) const {
131  return vertices[0] == rhs[0] &&
132  vertices[1] == rhs[1] &&
133  vertices[2] == rhs[2] &&
134  vertices[3] == rhs[3];
135  }
136 
137  int operator[](int idx) const { return vertices[idx]; }
138  int& operator[](int idx) { return vertices[idx]; }
139 };

146 // ----- 4PCS Options -----
147 // delta and overlap_estimation are the application parameters. All other
148 // parameters are more likely to keep fixed and they can be set via the setters.
149 struct Match4PCSOptions {
150  using Scalar = typename Point3D::Scalar;
151  Match4PCSOptions() {}
152 
153  // The delta for the LCP (see the paper).
154  Scalar delta = 5.0;
155 
156  // Maximum normal difference.
157  Scalar max_normal_difference = -1;
158  // Maximum translation distance. Set negative to ignore
159  Scalar max_translation_distance = -1;
160  // Maximum rotation angle. Set negative to ignore
161  Scalar max_angle = -1;
162  // Maximum color RGB distance between corresponding vertices. Set negative to ignore
163  Scalar max_color_distance = -1;
164  // The number of points in the sample. We sample this number of points
165  // uniformly from P and Q.
166  size_t sample_size = 200;
167  // Maximum time we allow the computation to take. This makes the algorithm
168  // an ANY TIME algorithm that can be stopped at any time, producing the best
169  // solution so far.
170  int max_time_seconds = 60;
171  // use a constant default seed by default
172  unsigned int randomSeed = std::mt19937::default_seed;
173 
174  inline bool configureOverlap(Scalar overlap_, Scalar terminate_threshold_ = Scalar(1)) {
175  if(terminate_threshold_ < overlap_) return false;
176  overlap_estimation = overlap_;
177  terminate_threshold = terminate_threshold_;
178  return true;
179  }
180  inline Scalar getTerminateThreshold() const { return terminate_threshold; }
181  inline Scalar getOverlapEstimation() const { return overlap_estimation; }
182 
183 private:
184  // Threshold on the value of the target function (LCP, see the paper).
185  // It is used to terminate the process once we reached this value.
186  Scalar terminate_threshold = 1.0;
187  // Estimated overlap between P and Q. This is the fraction of points in P that
188  // may have corresponding point in Q. It's being used to estimate the number
189  // of RANSAC iterations needed to guarantee small failure probability.
190  Scalar overlap_estimation = 0.2;
191 };
